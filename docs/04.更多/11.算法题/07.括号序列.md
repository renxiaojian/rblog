## 1 题目

给出一个仅包含字符'(',')','{','}','['和']',的字符串，判断给出的字符串是否是合法的括号序列
括号必须以正确的顺序关闭，"()"和"()[]{}"都是合法的括号序列，但"(]"和"([)]"不合法。

**示例**

```
输入："[965]4gf[dfg]sd8{{9t}5445[]}"
返回值：true
```

## 2 题解

```
/**
  * 
  * @param s string字符串 
  * @return bool布尔型
  */
function isValid( s ) {
    let exp = [
        ['(',')'],
        ['[',']'],
        ['{','}']
    ]
    const map = new Map(exp)
    const stack = []
    for (let i of s) {
        if (map.has(i)) {
            stack.push(i)
        } else {
            if ( i == ')' || i == ']' || i == '}') {
                let sEnd = stack[stack.length-1]
                if(stack.length===0) return false
                if(map.get(sEnd) === i){
                    stack.pop()
                } else return false
            }
        }
    }
    return stack.length === 0
}
module.exports = {
    isValid : isValid
};
```
